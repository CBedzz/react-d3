class Graph extends React.Component {
  constructor(props) {
    super(props);
    this.state = {selected: null};

    this.selectNode = this.selectNode.bind(this);
  }

  shouldComponentUpdate(nextProps, nextState) {
    if (nextProps.version === this.props.version) {
      // if version is the same, no updates to data
      // so it must be interaction to select+highlight a node
      var highlightedNodes = {};
      this.links.attr('opacity', (d) => {
        if (!nextState.selected) return 0.5;
        if (d.source.key === nextState.selected) {
          highlightedNodes[d.target.key] = 1;
          return 0.5;
        } else if (d.target.key === nextState.selected) {
          highlightedNodes[d.source.key] = 1;
          return 0.5;
        }
        return 0.1;
      });
      this.circles.attr('opacity', d => !nextState.selected ||
        nextState.selected === d.key || highlightedNodes[d.key] ? 1 : 0.2);

      return false;
    }
    return true;
  }

  componentDidMount() {
    this.container = d3.select(this.refs.container);
    this.calculateData();
    this.renderLinks();
    this.renderNodes();
  }

  componentDidUpdate() {
    this.calculateData();
    this.renderLinks();
    this.renderNodes();
  }

  calculateData() {
    var {nodes, links} = this.props;
    simulation.nodes(nodes)
      .force('link', d3.forceLink(links).id(d => d.key).distance(100));

    _.times(2000, () => simulation.tick());
    this.nodes = nodes;
    this.links = links;
  }

  renderNodes() {
    this.circles = this.container.selectAll('circle')
      .data(this.nodes, d => d.key);
    // exit
    this.circles.exit().remove();
    // enter + update
    this.circles = this.circles.enter().append('circle')
      .classed('node', true)
      .merge(this.circles)
      .attr('cx', d => d.x)
      .attr('cy', d => d.y)
      .attr('r', d => d.size)
      .on('click', this.selectNode);
  }

  renderLinks() {
    this.links = this.container.selectAll('line')
      .data(this.links, d => d.key);
    // exit
    this.links.exit().remove();
    // enter + update
    this.links = this.links.enter().insert('line', 'circle')
      .classed('link', true)
      .merge(this.links)
      .attr('stroke-width', d => d.size)
      .attr('x1', d => d.source.x)
      .attr('x2', d => d.target.x)
      .attr('y1', d => d.source.y)
      .attr('y2', d => d.target.y);
  }

  selectNode(node) {
    if (node.key === this.state.selected) {
      this.setState({selected: null});
    } else {
      this.setState({selected: node.key});
    }
  }

  render() {
    return (
      
    )
  }
}
